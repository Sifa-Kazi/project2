df['churn_probability'] = model.predict_proba(X)[:,1]

def segment(row):
    if row['churn_probability'] > 0.7:
        return 'At Risk'
    elif row['Account length'] > 120:
        return 'Loyal'
    else:
        return 'Dormant'
